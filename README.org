  : tt [{:id => 1, :name => "alice"}, {:id => 2, :name => "bob"}]
  : # >> +----+-------+
  : # >> | id | name  |
  : # >> +----+-------+
  : # >> |  1 | alice |
  : # >> |  2 | bob   |
  : # >> +----+-------+

  : tt({:id => 1, :name => "alice"})
